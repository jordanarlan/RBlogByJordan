genre_year <- data %>% 
  filter(Year_of_Release != c(2017, 2020, NA))%>%
  group_by(Genre, Year_of_Release) %>%
  summarise(Global_Sales_tot = sum(Global_Sales)) 

  ggplot(genre_year, aes(x = Year_of_Release, y = Global_Sales_tot, color = Genre)) +
